# Periodische Sensor-Polls (alle 5s). Verfuegbar:
#   ping, uptime, boot_time, temperatures, fans, display, is_muted,
#   easire, mpv_file_pos_sec
# easire setzt voraus dass der easire-Player als Prozess laeuft.
# mpv_file_pos_sec setzt voraus dass das mpv_control-Tool installiert ist.
PROBE_METHODS="ping,temperatures,fans,uptime,display,is_muted"

# Vom Manager akzeptierte Kommandos. Verfuegbar:
#   wake, shutdown, reboot, mute, unmute, ping
# 'wake' wird vom Manager extern via WoL ausgeloest, die Probe sieht es nie.
PROBE_CAPABILITIES="wake,shutdown,reboot,mute,unmute"

# Optional: explizite MQTT-Identitaet (client_id + Topic-Prefix
# probe/<id>/... , manager/<id>/+). Ueberschreibt socket.getfqdn().
# NUR setzen, wenn getfqdn() auf der Box den falschen Namen liefert
# (z.B. Multi-NIC/kein DNS-Suffix -> bare hostname statt <name>.<suffix>).
# Wert muss der vom Manager erwarteten FQDN entsprechen. Ungueltige Werte
# (Leerzeichen oder / + #) werden ignoriert -> Fallback auf getfqdn().
